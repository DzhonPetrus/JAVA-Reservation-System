#SELECT DB
USE RESERVATION_SYSTEM;

#CREATE ACCOUNTS TABLE
CREATE TABLE ACCOUNTS(ID INT PRIMARY KEY AUTO_INCREMENT NOT NULL, USERNAME VARCHAR(20) UNIQUE NOT NULL, PASSWORD VARCHAR(20) NOT NULL, PERSON_ID VARCHAR(20) UNIQUE NOT NULL, FNAME VARCHAR(25), LNAME VARCHAR(25), ROLE INT NOT NULL);

#CREATE ITEMS TABLE
CREATE TABLE ITEMS(ID INT PRIMARY KEY AUTO_INCREMENT NOT NULL, NAME VARCHAR(50) NOT NULL, STOCK INT NOT NULL);

#CREATE TRANSACTIONS TABLE
CREATE TABLE TRANSACTIONS(ID INT PRIMARY KEY AUTO_INCREMENT NOT NULL, USER_ID INT, BORROWER_ID VARCHAR(20) NOT NULL, BORROWER_FNAME VARCHAR(25), BORROWER_LNAME VARCHAR(25), ITEM_ID INT, STATUS VARCHAR(15), DATE_RES DATE, DATE_UPDATED TIMESTAMP);
